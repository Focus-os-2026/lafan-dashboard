เชื่อมต่อระบบจัดเก็บข้อมูล Google Sheets

เพื่อเพิ่มร้านฝากขายใหม่และเขียนข้อมูลกลับไปยัง Google Sheet ของโครงการโดยตรง Dashboard นี้ใช้ Google Apps Script Web App ในการซิงค์ข้อมูลอย่างปลอดภัย

ระบบจะบันทึกการตั้งค่านี้ไว้ในเบราว์เซอร์ของคุณ

🛠️ วิธีติดตั้ง Google Apps Script ใน Google Sheet ของคุณ

  1. เปิด Google Sheet ของคุณ
  2. ไปที่เมนู ส่วนขยาย (Extensions) > Apps Script
  3. คัดลอกโค้ดสคริปต์ด้านล่างนี้ ไปวางทับในเครื่องมือแก้ไขสคริปต์
    4. function doPost(e) { try { var params = JSON.parse(e.postData.contents); var sheetName = params.sheetName; var rowData = params.rowData; // Array ของข้อมูล var doc = SpreadsheetApp.getActiveSpreadsheet(); var sheet = doc.getSheetByName(sheetName); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "ไม่พบแท็บ: " + sheetName })) .setMimeType(ContentService.MimeType.JSON); } sheet.appendRow(rowData); return ContentService.createTextOutput(JSON.stringify({ status: "success" })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } }
  4. คลิก ทำให้ใช้งานได้ (Deploy) > การจัดการทำให้ใช้งานได้ใหม่ (New deployment)
  5. เลือกประเภทเป็น เว็บแอป (Web app)
  6. ตั้งค่าสิทธิ์การเข้าถึงเป็น:
    • เรียกใช้ในฐานะ: ฉัน (อีเมลของคุณ)
    • ผู้มีสิทธิ์เข้าถึง: ทุกคน (Anyone)
  7. คลิก ทำให้ใช้งานได้ (Deploy) และให้สิทธิ์เข้าถึงบัญชี
  8. คัดลอก URL เว็บแอป (Web app URL) ที่ได้ นำมาใส่ในช่องตั้งค่าด้านบนเป็นอันเสร็จสิ้น!
